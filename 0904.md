## 삼항연산자
```html
    <script>
      
      // 조건식 ? 실행문1 : 실행문2
      // 조건식 주어질 시 맞으면 실행문1 아니면 실행문2

      var num1 = 10;
      var num2 = -10;

      console.log(num1>0 ? '양수' : '음수');
      console.log(num2>0 ? '양수' : '음수');

    </script>
```

## 제어문 - 반복문
```html
    <script>

        /*
        
        for(초기값; 조건식; 증가감식 //끊어주기 위함으로서 끝에는 세미콜론을 안 붙여도 된다 )
        {
            실행문
        }
        
        */

        var sum = 0;
        
        for(var i = 1; i < 10; i++)
        {
            sum+=i;
        }

        console.log(sum);

    </script>
```

## 조건문 if
```html
    <script>
        
        /*
    
        1. 조건문 if문
        if(조건식)
        {
            실행문
        }
        
        */

        var age = 10;
        var admissonFree = '';

        if(age<13)
        {
            admissonFree = '2000';
        }

        console.log(admissonFree);

        //2. step

        var age1 = 20;
        var admissonFree1 = '';
        
        if(age1 < 13)
        {
            admissonFree1 = '2000';
        }
        else
        {
            admissonFree1 = '4000';
        }

        console.log(admissonFree1);

        //3. step

        var age2 = 20;
        var admissonFree2 = '';

        if(age2 < 13)
        {
            admissonFree2 = '2000원';
        }
        else if(age2 >= 13 && age2 < 20)
        {
            admissonFree2 = '4000원';
        }
        else
        {
            admissonFree2 = '5000원';
        }

        console.log(admissonFree2);


    </script>
```

## 조건문 - 중첩
```html
    <script>
      
        /*
        1. 중첩 조건문 : if문 안에 또 다른 if문을 중첩하여 사용할 수 있다.

        if(조건식)
        {
            if(조건식)
            {
                실행문
            }
        }
        */


        //prompt 함수 특징 : 입력받은 숫자를 문자로 인식, number 함수 이용해서 다시 숫자 데이터로 변경해준다.
        var html = Number(prompt('html점수를 입력하세요.', ''));

        var css = Number(prompt('css점수를 입력하세요.', ''));

        var javascript = Number(prompt('javascript점수를 입력하세요.', ''));

        var avg = (html + css + javascript)/3

        var comment;

        if(avg >= 60)
        {
            if(html < 40)
            {
                comment = 'html' + ' 40점 미만 불합격';
            }
            else if(css < 40)
            {
                comment = 'css' + ' 40점 미만 불합격';
            }
            else if(javascript < 40)
            {
                comment = 'javascript' + ' 40점 미만 불합격';
            }
            else
            {
                comment = '평균 ' + avg + '점 합격';
            }
        }
        else
        {
            comment = '평균 ' + avg + '점 불합격';
        }

        document.write(comment);

    </script>
```
