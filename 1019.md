### 문제 : OX 퀴즈
정답을 맞췄을 경우 문제당 1점, 연속적으로 맞출 경우 연속한 정답 개수 만큼 가산점 부여, 점수 산출 프로그램
```js
function answer(mark) {
    let result = 0;

    // 1 -> 1점, 연속한 1인 경우 연속한 count만큼 점수 추가
    let score = 0;
    for (let i = 0; i < mark.length; i++) {
        if (mark[i] == 1) {
            result += ++score;
        } else {
            score = 0;
        }
    }

    return result;
}
```
