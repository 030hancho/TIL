```css
*
{
    margin: 0;
    padding: 0;
}

body{background-color: #eee;}

#totalwrap
{
    width: 940px;
    margin: 0 auto;
}

#totalwrap .title
{
    padding: 50px 0;
    text-align: center;
}

.wrapper
{
    background-color: beige;
    overflow: hidden;
    padding: 20px 0;
}

.col
{
    float: left;
    border-radius: 15px;
    width: 300px;
    margin-right: 20px;
}

.last
{
    margin-right: 0 !important;
}

/* flip_hover */

.flip_hover_wrap
{
    width: 300px;
    height: 190px;
    position: relative;

    /* 3D효과를 주는 부모한테는 perspective를 준다. 숫자 : 돌아갈 수 있는 공간 */
    perspective: 1000px;

}


.front, .back
{
    width: inherit;
    height: inherit;
    border-radius: 15px;
    line-height: 190px;
    text-align: center;
    color: white;
    
    position: absolute;
    top: 0;
    left: 0;
    font-size: 30px;

    /* 3d 되는 요소에 선언 */
    transform-style: preserve-3d;

    backface-visibility: hidden;
    transition: all 1s ease;
}

.front
{
    background-color: skyblue;
    z-index: 2;
    transform: rotateX(0) rotateY(0);
}

.back
{
    background-color: tomato;
    transform: rotateY(-180deg);
}

.flip_hover_wrap:hover .front
{
    transform: rotateY(180deg);
}
.flip_hover_wrap:hover .back
{
    transform: rotateX(0) rotateY(0);
}

.box1, .box2{}

.box1{}

.box2{}
```
